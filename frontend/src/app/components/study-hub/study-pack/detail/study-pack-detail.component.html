    "correctAnswer": 1,
    "explanation": "Explication de la réponse...",
    "trapNote": "Piège fréquent...",
    "topic": "Sujet/Chapitre"
  },
  {
    "question": "Le T-SQL est une norme standard ANSI.",
    "type": "true-false",
    "correctAnswer": "False",
    "explanation": "Non, c'est une extension propriétaire de Microsoft.",
    "topic": "T-SQL"
  }
]
